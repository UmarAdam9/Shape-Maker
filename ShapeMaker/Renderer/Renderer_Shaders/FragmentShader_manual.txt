#version 330 core

out vec4 FragColor;
in vec4 color;

void main()
{
   float st = gl_FragCoord.y / 800.0f;

   if(st < 0.25)  //shouldnt this take half the screen??
   {
    FragColor = vec4(color.xyz,color.a*st/0.25 );
   }

   else
   {
    FragColor = vec4(color.xyza);
   }


}
